export class AppCustomerService {
    constructor(private readonly http: HttpClient, private readonly gridService: GridService) { }

    add = (customer: Customer) =&gt; this.http.post&lt;number&gt;("customers", customer);

    delete = (id: number) =&gt; this.http.delete(`customers/${id}`);

    get = (id: number) =&gt; this.http.get&lt;Customer&gt;(`customers/${id}`);

    grid = (parameters: GridParameters) =&gt; this.gridService.get&lt;Customer&gt;("customers/grid", parameters);

    inactivate = (id: number) =&gt; this.http.patch(`customers/${id}/inactivate`, {});

    list = () =&gt; this.http.get&lt;Customer[]&gt;("customers");

    update = (customer: Customer) =&gt; this.http.put(`customers/${customer.id}`, customer);
}
